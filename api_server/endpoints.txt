// restaurant routes
get -> '/restaurants', 'RestaurantController@index' -> list all restaurants
post -> '/restaurants', 'RestaurantController@store' -> add new restaurant
get -> '/restaurant/{id}', 'RestaurantController@show' -> show the restaurant with specified id
post -> '/restaurant/{id}', 'RestaurantController@edit' -> edit the restaurant with specified id

// table routes
get -> '/restaurant/{id}/tables', 'TableController@index' -> list tables that belongs to restaurant
post -> '/restaurant/{id}/tables', 'TableController@store' -> add new table to restaurant
get -> '/restaurant/{id}/table/{tableId}', 'TableController@show' ->
post -> '/restaurant/{id}/table/{tableId}', 'TableController@edit'
